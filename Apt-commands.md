```
# Bijwerken van apt repository
sudo apt update

# Upgrade van packages die dit nodig hebben
sudo apt upgrade

# Downloaden van *.deb package voor opslag/archivering
apt-get download

# Download locatie is de directory waarin het commando is uitgevoerd.
# Download locatie voor overige apt systeem installaties (apt upgrade): 
# /var/cache/apt/archives	

# clean your whole cache
sudo apt-get clean

# verwijderen van gedownloade apt packages (opschonen)
sudo apt purge 

# Clear the cache of any software that's currently not installed
sudo apt-get autoclean


```
